CREATE TABLE CARD (
 CARD_ID INTEGER PRIMARY KEY,
 PASSENGER_ID INTEGER NOT NULL,
 BALANCE INTEGER DEFAULT '0.00'
);

CREATE TABLE EMPLOYEE (
 EMPLOYEE_ID INT PRIMARY KEY ,
 EMPLOYEE_NAME TEXT NOT NULL,
 JOB TEXT CHECK(JOB = 'STATION MASTER' OR JOB = 'DRIVER' OR JOB = 'SUPERVISER' OR JOB = 'MECHANIC'),
 CONTACT_NO INT UNIQUE,
 SALARY INT
);

CREATE TABLE PASSENGER (
 PASSENGER_ID INT,
 PASSENGER_NAME TEXT NOT NULL,
 CARD_ID INT,
 SEX TEXT CHECK(SEX = 'MALE' OR SEX = 'FEMALE'),
 CONTACT_NO INT UNIQUE,
 NID INT UNIQUE,
 PRIMARY KEY(PASSENGER_ID,CARD_ID)
);

CREATE TABLE STATION_LIST (
 STATION_ID INT PRIMARY KEY ,
 STATION_NAME TEXT NOT NULL,
 EMPLOYEE_ID INT UNIQUE
);

CREATE TABLE TRAIN_INFO (
 TRAIN_ID INT PRIMARY KEY ,
 TOTAL_SEAT INT,
 AVAILABLE_SEAT INT,
 EMPLOYEE_ID_01 INT UNIQUE,
 EMPLOYEE_ID_02 INT UNIQUE
);


CREATE TABLE TRAIN_MAINTANACE (
 MAINTANANCE_NO INT PRIMARY KEY ,
 TRAIN_ID INT,
 EMPLOYEE_ID INT,
 DESCRIPTION TEXT,
 COST INT DEFAULT 0.00,
 DATE TEXT
);


CREATE TABLE TRAVEL_INFO (
 TRAVEL_NO INT PRIMARY KEY ,
 PASSENGER_ID INT,
 TRAIN_ID INT,
 STATION_A TEXT CHECK(STATION_A <> STATION_B),
 STATION_B TEXT CHECK(STATION_A <> STATION_B),
 PAID_AMOUNT NUMERIC,
 DATE TEXT,
 TIME TEXT
);
